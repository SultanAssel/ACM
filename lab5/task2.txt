class Solution {
	

	

	public:
	

	   ListNode* reverseList(ListNode* head) {
	

	       if(head == NULL)
	

	           return head;
	

	      
	

	       ListNode * current = head;
	

	       ListNode * prev;
	

	       ListNode * next = NULL;
	

	      
	

	       while(true) {
	

	           prev = current->next;
	

	           current->next = next;
	

	           next = current;
	

	           if(prev != NULL)
	

	               current = prev;
	

	           else
	

	               break;
	

	       }
	

	       return current;
	

	   }
	

	};
